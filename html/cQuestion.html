#include<stdio.h>
              
int decimalToBinary(int num)
{
    int res = 0;
    int i = 1;

    while(num > 0)
    {
        int r = num % 2;
        num /= 2;

        res = res + r * i;
        i *= 10;

    }

    return res;

}

int main()
{
    int fnum , snum;
    printf("Enter the First Number.\n");
    scanf("%d",&fnum);
    printf("Enter the Second Number.\n");
    scanf("%d",&snum);

    int count = 0;
    int bfnum = decimalToBinary(fnum);
    int bsnum = decimalToBinary(snum);

    printf("Binary Of %d = %d\n",fnum , bfnum);
    printf("Binary Of %d = %d\n",snum , bsnum);

    while(bfnum != 0 || bsnum != 0)
    {
        int fnbit = bfnum > 0 ? bfnum % 10 : 0;
        bfnum /= 10;
        int snbit = bsnum > 0 ? bsnum % 10 : 0;
        bsnum /= 10;

        if(fnbit != snbit)
        {
            count++;
        }
    }

    printf("%d bits.\n",count);
}

        if(fnum % i == 0 && snum % i == 0)
