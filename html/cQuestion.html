    // Code for Reversing a num.

    int temp = num;
    while(temp != 0)
    {
        int lastDigit = temp % 10;
        temp /= 10;
        revNum = revNum * 10 + lastDigit;
    }

//    reverse of the number equal to actual number then the number is Palindrome.

    if(revNum == num)
